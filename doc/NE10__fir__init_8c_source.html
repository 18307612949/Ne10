52 ne10_result_t ne10_fir_init_float (ne10_fir_instance_f32_t * S,
53  ne10_uint16_t numTaps,
54  ne10_float32_t * pCoeffs,
55  ne10_float32_t * pState,
56  ne10_uint32_t blockSize)
57 {
58  /* Assign filter taps */
59  S->numTaps = numTaps;
60 
61  /* Assign coefficient pointer */
62  S->pCoeffs = pCoeffs;
63 
64  /* Clear state buffer and the size of state buffer is (blockSize + numTaps - 1) */
65  memset (pState, 0, (numTaps + (blockSize - 1u)) * sizeof (ne10_float32_t));
66 
67  /* Assign state pointer */
68  S->pState = pState;
69  return NE10_OK;
70 }

103  ne10_result_t status;
104 
105  /* The size of the input block must be a multiple of the decimation factor */
106  if ( (blockSize % M) != 0u)

110  }
111  else
112  {
113  /* Assign filter taps */
114  S->numTaps = numTaps;
115 
116  /* Assign coefficient pointer */
117  S->pCoeffs = pCoeffs;
118 
119  /* Clear state buffer and size is always (blockSize + numTaps - 1) */
120  memset (pState, 0, (numTaps + (blockSize - 1u)) * sizeof (ne10_float32_t));
121 
122  /* Assign state pointer */
123  S->pState = pState;
124 
125  /* Assign Decimation Factor */
126  S->M = M;
127 
128  status = NE10_OK;
129  }
130 
131  return (status);

159 ne10_result_t ne10_fir_interpolate_init_float (

164  ne10_float32_t * pState,
165  ne10_uint32_t blockSize)
166 {
167  ne10_result_t status;
168 
169  /* The filter length must be a multiple of the interpolation factor */
170  if ( (numTaps % L) != 0u)
171  {
172  /* Set status as NE10_ERR */
173  status = NE10_ERR;
174  }
175  else
176  {
177 
178  /* Assign coefficient pointer */
179  S->pCoeffs = pCoeffs;
180 
181  /* Assign Interpolation factor */
182  S->L = L;
183 
184  /* Assign polyPhaseLength */
185  S->phaseLength = numTaps / L;
186 
187  /* Clear state buffer and size of state array is always phaseLength + blockSize - 1 */
188  memset (pState, 0,
189  (blockSize +
190  ( (ne10_uint32_t) S->phaseLength - 1u)) * sizeof (ne10_float32_t));
191 
192  /* Assign state pointer */
193  S->pState = pState;
194 
195  status = NE10_OK;
196  }

198  return (status);
