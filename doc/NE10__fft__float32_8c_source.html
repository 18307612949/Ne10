00853 /* factors buffer:
00854  * 0: stage number
00855  * 1: stride for the first stage
00856  * others: factors */
00857 static ne10_int32_t ne10_factor (ne10_int32_t n, ne10_int32_t * facbuf)
00858 {
00859     ne10_int32_t p = 4;
00860     ne10_int32_t i = 1;
00861     ne10_int32_t stage_num = 0;
00862     ne10_int32_t stride_max = n;
00863 
00864     /* factor out powers of 4, powers of 2 */
00865     do
00866     {
00867         if ( (n % p) == 2)
00868             p = 2;
00869         else if (n % p)
00870         {
00871             return NE10_ERR;
00872         }
00873 
00874         n /= p;
00875         facbuf[2 * i] = p;
00876         facbuf[2 * i + 1] = n;
00877         i++;
00878         stage_num++;
00879     }
00880     while (n > 1);
00881     facbuf[0] = stage_num;
00882     facbuf[1] = stride_max / p;
00883     return NE10_OK;
00884 }

00887 static void ne10_fft_split_r2c_1d_float32 (ne10_fft_cpx_float32_t *dst,
00888         const ne10_fft_cpx_float32_t *src,
00889         ne10_fft_cpx_float32_t *twiddles,
00890         ne10_int32_t ncfft)
00891 {
00892     ne10_int32_t k;
00893     ne10_fft_cpx_float32_t fpnk, fpk, f1k, f2k, tw, tdc;
00894 
00895     tdc.r = src[0].r;
00896     tdc.i = src[0].i;
00897 
00898     dst[0].r = tdc.r + tdc.i;
00899     dst[ncfft].r = tdc.r - tdc.i;
00900     dst[ncfft].i = dst[0].i = 0;
00901 
00902     for (k = 1; k <= ncfft / 2 ; ++k)
00903     {
00904         fpk    = src[k];
00905         fpnk.r =   src[ncfft - k].r;
00906         fpnk.i = - src[ncfft - k].i;
00907 
00908         f1k.r = fpk.r + fpnk.r;
00909         f1k.i = fpk.i + fpnk.i;
00910 
00911         f2k.r = fpk.r - fpnk.r;
00912         f2k.i = fpk.i - fpnk.i;
00913 
00914         tw.r = f2k.r * (twiddles[k - 1]).r - f2k.i * (twiddles[k - 1]).i;
00915         tw.i = f2k.r * (twiddles[k - 1]).i + f2k.i * (twiddles[k - 1]).r;
00916 
00917         dst[k].r = (f1k.r + tw.r) * 0.5f;
00918         dst[k].i = (f1k.i + tw.i) * 0.5f;
00919         dst[ncfft - k].r = (f1k.r - tw.r) * 0.5f;
00920         dst[ncfft - k].i = (tw.i - f1k.i) * 0.5f;
00921     }
00922 }

01033 ne10_fft_cfg_float32_t ne10_fft_alloc_c2c_float32 (ne10_int32_t nfft)
01034 {

01036     ne10_uint32_t memneeded = sizeof (ne10_fft_state_float32_t)

01040                               + NE10_FFT_BYTE_ALIGNMENT;     /* 64-bit alignment*/

01050         st->buffer = st->twiddles + nfft;
