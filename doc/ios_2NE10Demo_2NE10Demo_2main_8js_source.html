40 // Add and show for a feature block
41 // argv[1:] be the bars
42 function addShowBarBlock(title) {
43  var i;
44  var max = 0;
45  var eBlock;
46  var eBar;
47 
48  for (i = 1; i < arguments.length; ++i) {
49  if (max < arguments[i]) {
50  max = arguments[i];
51  }
52  }
53  max *= 1.2; // this is for right padding
54 
55  // block is the container
56  eBlock = $("<div class='block'/>");
57 
58  $("<h3/>").html(title).appendTo(eBlock);
59  for (i = 1; i < arguments.length; ++ i) {
60  eBar = $("<div class='bar'/>");
61  eBar.css("width", 100 * arguments[i] / max + "%" );
62  eBar.css("background-color", barColors[i - 1]);
63  eBar.css("box-shadow", "5px 5px 15px " + barColors[i - 1]);
64  eBar.html(arguments[i]);
65 
66  eBar.appendTo(eBlock);
67  }
68 
69  eBlock.appendTo($("#content"));
70 }

90 function returnFromObjectivec(ne10TestResultStr)
91 {
92  var progress = 0;
93  var progress_display = 0;
94 
95  if (ne10TestResultStr) {
96  //parse json data from string
97  var testResult = eval( '(' + ne10TestResultStr + ')' );
98  for (i = 0; i < testResult.length; ++i) {
99  ++case_count;
100  testResult[i].name = case_count.toString() + " "
101  + testResult[i].name;
102  addShowBarBlock(testResult[i].name,
103  testResult[i].time_c,
104  testResult[i].time_neon);
105  acceleration_rate = acceleration_rate +
106  testResult[i].time_c / testResult[i].time_neon;
107  }
108 
109  progress = case_count/117*100;
110  progress_display = Math.floor(progress);
111  var font_height = $("#progress_bar_in").height()*0.7;
112 
113  $("#progress_bar_in").css("width", progress + "%");
114  $("#progress_bar_in").css("font-size",font_height + "px");
115  $("#progress_bar_in").html(progress_display + "%");
116 
117  //TODO
118  //Currently, 117 is number of total test cases.
119  //due to the limit of current test framework, we can't
120  //get the number of all test cases at compiling time.
121  if( case_count == 117 )
122  add_result(acceleration_rate, case_count);
123  }
124 }
