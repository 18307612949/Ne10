00123 static void ne10_split_rfft_float_c(
00124                      ne10_float32_t * pSrc,
00125                      ne10_uint32_t N,
00126                      ne10_float32_t * pReTable,
00127                      ne10_float32_t * pImTable,
00128                      ne10_float32_t * pDst)
00129 {
00130     ne10_uint32_t k;                                    /* Loop Counter */
00131     ne10_float32_t uRe,vRe,uIm,vIm;                    /* Temporary variables for output */
00132     ne10_float32_t reTwd,imTwd,reTmp,imTmp;
00133     ne10_float32_t *pCoefRe,*pCoefIm;                             /* Temporary pointers for twiddle factors */
00134     ne10_uint32_t NBy2 = N>>1;
00135 
00136     pCoefRe = pReTable;
00137     pCoefIm = pImTable;
00138 
00139     /*First Result*/
00140     pDst[0] = pSrc[0] + pSrc[1];
00141     pDst[1] = 0;
00142     /*N/2 th Result*/
00143     pDst[N] = pSrc[0] - pSrc[1];
00144     pDst[N+1] = 0;
00145 
00146     /*for k=1 to N/4-1 and k=N/4+1 to K=N/2-1*/
00147     for(k=1;k<(N>>2);k++)
00148     {
00149         /*uRe = (a[k]+a[N/2-k])/2*/
00150         uRe = (pSrc[2*k]+pSrc[2*(NBy2-k)])*0.5;
00151         /*uIm = (b[k]-b[N/2-k])/2*/
00152         uIm = (pSrc[2*k+1]-pSrc[2*(NBy2-k)+1])*0.5;
00153         /*VRe = (b[k]+b[N/2-k])/2*/
00154         vRe = (pSrc[2*k+1]+pSrc[2*(NBy2-k)+1])*0.5;
00155         /*Vim = -(a[k]-a[N/2-k])/2*/
00156         vIm = (pSrc[2*(NBy2-k)]-pSrc[2*k])*0.5;
00157         reTwd = pCoefRe[k];
00158         imTwd = pCoefIm[k];
00159         reTmp = vRe*reTwd + vIm*imTwd;
00160         imTmp = vIm*reTwd - vRe*imTwd;
00161         pDst[2*k] = uRe + reTmp;
00162         pDst[2*k+1] = uIm + imTmp;
00163         pDst[2*(NBy2-k)] = uRe-reTmp;
00164         pDst[2*(NBy2-k)+1] = imTmp-uIm;
00165 
00166         /*Out Put from K=N/2+1 till k=N-1*/
00167         /*y[N-k] = conjugate(y[k] k=0 to k<N/2)*/
00168         pDst[2*(N-k)] = uRe + reTmp;
00169         pDst[2*(N-k)+1] = -(uIm + imTmp);
00170         pDst[2*(NBy2+k)] = uRe-reTmp;
00171         pDst[2*(NBy2+k)+1] = uIm-imTmp;
00172     }
00173     /*y[N/4] = a[N/4]-jb[N/4]; y[3*N/4] = a[N/4] + jb[N/4]*/
00174     pDst[NBy2] = pSrc[NBy2];
00175     pDst[NBy2+1] = -pSrc[NBy2+1];
00176     pDst[N+NBy2] = pSrc[NBy2];
00177     pDst[N+NBy2+1] = pSrc[NBy2+1];
00178 
00179 }

00223         imTmp = (pSrc[2*k+1] + pSrc[2*(NBy2-k)+1])*0.5;
00224

00226         imTwd = pCoefIm[k];
00227 
00228         /*VRe = (b[k]+b[N/2-k])/2*/
