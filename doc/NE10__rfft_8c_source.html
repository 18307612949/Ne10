124 static void ne10_split_rfft_float_c(
125  ne10_float32_t * pSrc,
126  ne10_uint32_t N,
127  ne10_float32_t * pReTable,
128  ne10_float32_t * pImTable,
129  ne10_float32_t * pDst)
130 {
131  ne10_uint32_t k; /* Loop Counter */
132  ne10_float32_t uRe,vRe,uIm,vIm; /* Temporary variables for output */
133  ne10_float32_t reTwd,imTwd,reTmp,imTmp;
134  ne10_float32_t *pCoefRe,*pCoefIm; /* Temporary pointers for twiddle factors */
135  ne10_uint32_t NBy2 = N>>1;
136 
137  pCoefRe = pReTable;
138  pCoefIm = pImTable;
139 
140  /*First Result*/
141  pDst[0] = pSrc[0] + pSrc[1];
142  pDst[1] = 0;
143  /*N/2 th Result*/
144  pDst[N] = pSrc[0] - pSrc[1];
145  pDst[N+1] = 0;
146 
147  /*for k=1 to N/4-1 and k=N/4+1 to K=N/2-1*/
148  for(k=1;k<(N>>2);k++)
149  {
150  /*uRe = (a[k]+a[N/2-k])/2*/
151  uRe = (pSrc[2*k]+pSrc[2*(NBy2-k)])*0.5;
152  /*uIm = (b[k]-b[N/2-k])/2*/
153  uIm = (pSrc[2*k+1]-pSrc[2*(NBy2-k)+1])*0.5;
154  /*VRe = (b[k]+b[N/2-k])/2*/
155  vRe = (pSrc[2*k+1]+pSrc[2*(NBy2-k)+1])*0.5;
156  /*Vim = -(a[k]-a[N/2-k])/2*/
157  vIm = (pSrc[2*(NBy2-k)]-pSrc[2*k])*0.5;
158  reTwd = pCoefRe[k];
159  imTwd = pCoefIm[k];
160  reTmp = vRe*reTwd + vIm*imTwd;
161  imTmp = vIm*reTwd - vRe*imTwd;
162  pDst[2*k] = uRe + reTmp;
163  pDst[2*k+1] = uIm + imTmp;
164  pDst[2*(NBy2-k)] = uRe-reTmp;
165  pDst[2*(NBy2-k)+1] = imTmp-uIm;
166 
167  /*Out Put from K=N/2+1 till k=N-1*/
168  /*y[N-k] = conjugate(y[k] k=0 to k<N/2)*/
169  pDst[2*(N-k)] = uRe + reTmp;
170  pDst[2*(N-k)+1] = -(uIm + imTmp);
171  pDst[2*(NBy2+k)] = uRe-reTmp;
172  pDst[2*(NBy2+k)+1] = uIm-imTmp;
173  }
174  /*y[N/4] = a[N/4]-jb[N/4]; y[3*N/4] = a[N/4] + jb[N/4]*/
175  pDst[NBy2] = pSrc[NBy2];
176  pDst[NBy2+1] = -pSrc[NBy2+1];
177  pDst[N+NBy2] = pSrc[NBy2];
178  pDst[N+NBy2+1] = pSrc[NBy2+1];
179 
180 }

195 static void ne10_split_rifft_float_c(
196  ne10_float32_t * pSrc,
197  ne10_uint32_t N,
198  ne10_float32_t * pReTable,
199  ne10_float32_t * pImTable,
200  ne10_float32_t * pDst)
201 {
202  ne10_uint32_t k; /* Loop Counter */
203  ne10_float32_t uRe,vRe,uIm,vIm; /* Temporary variables for output */
204  ne10_float32_t reTwd,imTwd,reTmp,imTmp;
205  ne10_float32_t *pCoefRe,*pCoefIm; /* Temporary pointers for twiddle factors */
206  ne10_uint32_t NBy2 = N>>1;
207 
208  pCoefRe = pReTable;
209  pCoefIm = pImTable;
210 
211  /*First Result*/
212  pDst[0] = (pSrc[0] + pSrc[N])*0.5;
213  pDst[1] = (pSrc[0] - pSrc[N])*0.5;
214 
215  /*for k=1 to N/4-1 and k=N/4+1 to K=N/2-1*/
216  for(k=1;k< (N>>2);k++)
217  {
218  /*uRe = (a[k]+a[N/2-k])/2*/
219  uRe = (pSrc[2*k] + pSrc[2*(NBy2-k)])*0.5;
220  /*uIm = (b[k]-b[N/2-k])/2*/
221  uIm = (pSrc[2*k+1] - pSrc[2*(NBy2-k)+1])*0.5;
222 
223  reTmp = (pSrc[2*k] - pSrc[2*(NBy2-k)])*0.5;
224  imTmp = (pSrc[2*k+1] + pSrc[2*(NBy2-k)+1])*0.5;
225 
226  reTwd = pCoefRe[k];
227  imTwd = pCoefIm[k];
228 
229  /*VRe = (b[k]+b[N/2-k])/2*/
230  vRe = reTmp*reTwd - imTmp*imTwd;
231  /*Vim = -(a[k]-a[N/2-k])/2*/
232  vIm = imTmp*reTwd + reTmp*imTwd;
233  pDst[2*k] = (uRe-vIm);
234  pDst[2*k+1] = (uIm+vRe);
235 
236  pDst[2*(NBy2-k)] = (uRe+vIm);
237  pDst[2*(NBy2-k)+1] = (vRe-uIm);
238  }
239  /*y[N/4] = a[N/4]-jb[N/4]*/
240  pDst[NBy2] = pSrc[NBy2];
241  pDst[NBy2+1] = -pSrc[NBy2+1];
242 
243 }
