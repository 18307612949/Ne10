00044 // This macro helps measure the performance of the code passed to it through the "code" argument

00046 #define MEASURE(res, code) \
00047    { \
00048     gettimeofday (&before, &zone); \
00049       code \
00050     gettimeofday (&after, &zone); \
00051     if (before.tv_usec > after.tv_usec) \
00052     { \
00053       after.tv_usec += 1000000; \
00054       after.tv_sec--; \
00055     } \
00056     lapsed.tv_usec = after.tv_usec - before.tv_usec; \
00057     lapsed.tv_sec  = after.tv_sec  - before.tv_sec; \
00058     res = lapsed.tv_sec + ((double)lapsed.tv_usec / 1000000.0); \
00059    }

00065 // Group 1 = Functions that take a dst, a src, and a cst ("DstSrcCst" for short)

00074 // The naming convention used in the following macros is as follows:
00075 //   SNAPP_<A>_OPERATION_<T>_<I>
00076 //   where
00077 //   <A> Stands for the title of the operation (add, mul, etc) followed by its type (C = const as in addc).
00078 //       The letter X - if used - means any such operation.
00079 //   <T> Indicates the type of the operation (float, vec2, etc.)
00080 //       The letter X - is used - means any type.
00081 //   <I> This indicates the implementation (it can be C, ASM, or NEON).
