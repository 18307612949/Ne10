57  ne10_int32_t *factors,
58  ne10_int32_t nfft,
59  ne10_int32_t inverse_fft)
60 {
61  if (fin == fout)
62  {
63  /* NOTE: for an in-place FFT algorithm. It just performs an out-of-place FFT into a temp buffer */
64  ne10_fft_cpx_int32_t * tmpbuf_ = (ne10_fft_cpx_int32_t*) NE10_MALLOC (sizeof (ne10_fft_cpx_int32_t) * nfft);
65 
66  // copy the data from input to output and bit reversal
67  ne10_data_bitreversal_int32 (tmpbuf_, fin, 1, &factors[2]);
68 
69  if (inverse_fft)
70  ne10_mixed_radix_butterfly_inverse_int32_neon (tmpbuf_, factors, twiddles);
71  else
72  ne10_mixed_radix_butterfly_int32_neon (tmpbuf_, factors, twiddles);
73 
74  memcpy (fout, tmpbuf_, sizeof (ne10_fft_cpx_int32_t) *nfft);
75  NE10_FREE (tmpbuf_);
76  }
77  else
78  {
79  // copy the data from input to output and bit reversal
80  ne10_data_bitreversal_int32 (fout, fin, 1, &factors[2]);
81 
82  if (inverse_fft)
83  ne10_mixed_radix_butterfly_inverse_int32_neon (fout, factors, twiddles);
84  else
85  ne10_mixed_radix_butterfly_int32_neon (fout, factors, twiddles);
86  }
87 }
