142  /* fN(n) = x(n) */
143  fcurr = *pSrc++;

157  /* Process sample for first tap */
158  gcurr = *px1++;
159  /* fN-1(n) = fN(n) - kN * gN-1(n-1) */
160  fnext = fcurr - ( (*pk) * gcurr);
