18 ## FSPECIAL - Create spatial filters for image processing
19 ##
20 ## Usage: f = fspecial(filtertype, optionalparams)
21 ##
22 ## filtertype can be
23 ##
24 ## 'average' - Rectangular averaging filter
25 ## 'disk' - Circular averaging filter.
26 ## 'gaussian' - Gaussian filter.
27 ## 'log' - Laplacian of Gaussian.
28 ## 'laplacian' - Crude 3x3 spproximation to laplacian
29 ## 'unsharp' - Crude sharpening filter
30 ## 'motion' - Motion blur filter of width 1 pixel
31 ## 'sobel' - Horizontal Sobel edge filter
32 ## 'prewitt' - Horizontal Prewitt edge filter
33 ##
34 ## The parameters that need to be specified depend on the filtertype
35 ##
36 ## Examples of use and associated default values:
37 ##
38 ## f = fspecial('average',sze) sze can be a 1 or 2 vector
39 ## default is [3 3].
40 ## f = fspecial('disk',radius) default radius = 5
41 ## f = fspecial('gaussian',sze, sigma) default sigma is 0.5
42 ## f = fspecial('laplacian',sze, sigma) default sze is [5 5]
43 ## default sigma is 0.5
44 ## f = fspecial('log');
45 ## f = fspecial('motion', length, angle) default length is 9
46 ## default angle is 0 (degrees)
47 ## f = fspecial('sobel');
48 ## f = fspecial('prewitt');
49 ##
50 ## Where sze is specified as a single value the filter will be square.
51 
52 ## Author: Peter Kovesi <pk@csse.uwa.edu.au>
53 ## Keywords: image processing, spatial filters
54 ## Created: August 2005

57 function f = fspecial(varargin)
58 
59  [filtertype, sze, sigma, radius, len, angle] = checkargs(varargin(:));
60 
61  rows = sze(1); cols = sze(2);
62  r2 = (rows-1)/2; c2 = (cols-1)/2;
63 
64  if strcmpi(filtertype, 'average')
65  f = ones(sze)/(rows*cols);
66 
67  elseif strcmpi(filtertype, 'disk')
68  [x,y] = meshgrid(-c2:c2, -r2:r2);
69  rad = sqrt(x.^2 + y.^2);
70  f = rad <= radius;
71  f = f/sum(f(:));
72 
73  elseif strcmpi(filtertype, 'gaussian')
74  [x,y] = meshgrid(-c2:c2, -r2:r2);
75  radsqrd = x.^2 + y.^2;
76  f = exp(-radsqrd/(2*sigma^2));
77  ##f = f/sum(f(:));
78 
79  elseif strcmpi(filtertype, 'log')
80  [x,y] = meshgrid(-c2:c2, -r2:r2);
81  radsqrd = x.^2 + y.^2;
82  f = -1/(pi*sigma^4)*(1-radsqrd/(2*sigma^2))...
83  .*exp(-radsqrd/(2*sigma^2));
84  f = f-mean(f(:)); # Ensure 0 DC
85 
86  elseif strcmpi(filtertype, 'laplacian')
87  f = [1 1 1
88  1 -8 1
89  1 1 1];
90 
91  elseif strcmpi(filtertype, 'unsharp')
92  f = -fspecial('log') + [0 0 0
93  0 1 0
94  0 0 0];
95 
96  elseif strcmpi(filtertype, 'sobel')
97  f = [1 2 1; 0 0 0; -1 -2 -1];
98 
99  elseif strcmpi(filtertype, 'prewitt')
100  f = [1 1 1; 0 0 0; -1 -1 -1];
101 
102  elseif strcmpi(filtertype, 'motion')
103  # First generate a horizontal line across the middle
104  f = zeros(sze);
105  f(floor(len/2)+1,1:len) = 1;
106 
107  # Then rotate to specified angle
108  f = imrotate(f,angle,'bilinear','loose');
109  f = f/sum(f(:));
110 
111  else
112  error('Unrecognized filter type');
113 
114  end
